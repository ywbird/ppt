import { file, write } from "bun"; // 템플릿 내부에 {{TITLE}}, {{THEME}}, {{HIGHLIGHT}}, {{SLIDES}} 플레이스홀더를 사용합니다. const HTML_TEMPLATE = ` {{TITLE}}
{{SLIDES}}
`; async function buildPresentation() { const args = Bun.argv.slice(2); if (args.length < 2) { console.error("❌ Error: Please specify both input and output file paths."); console.error("👉 Usage: bun run build.ts "); process.exit(1); } const markdownPath = args[0]; const outputPath = args[1]; try { const mdContent = await file(markdownPath).text(); let content = mdContent.trim(); let frontmatter = ""; // 기본값 설정 let pageTitle = "reveal.js"; let theme = "black"; // reveal.js의 기본 테마는 black입니다. let highlight = "tokyo-night-dark"; // 1. Frontmatter 추출 및 옵션 파싱 const fmRegex = /^\s*---\n([\s\S]*?)\n---\n/; const match = content.match(fmRegex); if (match) { frontmatter = match[1]; content = content.replace(fmRegex, ""); // Title 추출 const titleMatch = frontmatter.match(/^title:\s*(.+)$/im); if (titleMatch) pageTitle = titleMatch[1].trim().replace(/^["']|["']$/g, ''); // Theme 추출 const themeMatch = frontmatter.match(/^theme:\s*(.+)$/im); if (themeMatch) { let parsedTheme = themeMatch[1].trim().replace(/^["']|["']$/g, '').toLowerCase(); // 사용자가 default라고 적으면 black으로 변환 theme = parsedTheme === 'default' ? 'black' : parsedTheme; } // Highlight 추출 const highlightMatch = frontmatter.match(/^highlight:\s*(.+)$/im); if (highlightMatch) { let parsedHighlight = highlightMatch[1].trim().replace(/^["']|["']$/g, '').toLowerCase(); // 사용자가 tokyo라고 적으면 실제 파일명인 tokyo-night-dark로 변환 highlight = parsedHighlight === 'tokyo' ? 'tokyo-night-dark' : parsedHighlight; } console.log(`[Config] Title: ${pageTitle} | Theme: ${theme} | Highlight: ${highlight}`); } // 2. 슬라이드 파싱 및 HTML 렌더링 (Bun 내장 Markdown 파서 사용) const horizontalSlides = content.split(/\n\s*---\s*\n/); let slidesHtml = ""; for (const hSlide of horizontalSlides) { const verticalSlides = hSlide.split(/\n\s*\*\*\*\s*\n/); if (verticalSlides.length > 1) { slidesHtml += `\n
`; for (const vSlide of verticalSlides) { const renderedHtml = Bun.markdown.html(vSlide.trim()); slidesHtml += `\n
\n${renderedHtml}\n
`; } slidesHtml += `\n
`; } else { const renderedHtml = Bun.markdown.html(hSlide.trim()); slidesHtml += `\n
\n${renderedHtml}\n
`; } } // 3. 템플릿의 플레이스홀더 치환 const finalHtml = HTML_TEMPLATE .replace('{{TITLE}}', pageTitle) .replace('{{THEME}}', theme) .replace('{{HIGHLIGHT}}', highlight) .replace('{{SLIDES}}', slidesHtml); // 4. 저장 await write(outputPath, finalHtml); console.log(`✅ Build successful! -> ${outputPath}`); } catch (error) { console.error("❌ An error occurred during the build. Please check the file paths.", error); } } buildPresentation();
